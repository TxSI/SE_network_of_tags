post_filter = {'@CreationDate' : {'$gt':'2019'}}
posts = dbm.posts.find(post_filter)
all_tags, all_locations = {}, {}
location_nan, tags_nan = 0, 0
for post in posts:
    if '@OwnerLocation' in post:
        location = post['@OwnerLocation']
        if location in all_locations:
            all_locations[location] += 1
        else:
            all_locations[location] = 1
    else:
        location_nan += 1
    if '@Tags' in post:
        tags = post['@Tags']
        for tag in tags:
            if tag in all_tags and type(all_tags) != type("") and len(str(tag))>2:
                all_tags[tag] += 1
            else:
                all_tags[tag] = 1
    else:
        tags_nan += 1

dbm = pymongo.MongoClient()['stackexchange_demo']

def plot_tag_pop(tag="", location_a="", location_b=""):
    post_filter = {'@CreationDate' : {'$gt':'2019'},
                   '@Tags' : tag,
                   '$or' : [{'@OwnerLocation':{'$regex':'.*'+location_a+'.*'}}, 
                            {'@OwnerLocation':{'$regex':'.*'+location_b+'.*'}}]}
    posts = list(dbm.posts.find(post_filter))
    print("Posts in total:", len(posts))
    max_weeks = 22
    dist_a = [0 for _ in range(max_weeks)] 
    dist_b = [0 for _ in range(max_weeks)] 
    for post in posts:
        creation_date = datetime.datetime.strptime(post['@CreationDate'].split('T')[0], '%Y-%m-%d')
        creation_week = creation_date.date().isocalendar()[1]
        
        if post['@OwnerLocation'].__contains__(location_a):
            dist_a[creation_week -1] += 1
        else:
            dist_b[creation_week -1] += 1

    labels = ["w{}".format(x) for x in range(1, max_weeks+1)]

    x = np.arange(len(labels))  # the label locations
    width = 0.45  # the width of bars
    
    fig, ax = plt.subplots()
    fig.set_figheight(5)
    fig.set_figwidth(15)
    rects1 = ax.bar(x - width/2, dist_a, width, label=location_a)
    rects2 = ax.bar(x + width/2, dist_b, width, label=location_b)

    # Adding labels
    ax.set_ylabel('# of posts')
    ax.set_title("Posts related to the tag '{}'".format(tag))
    ax.set_xticks(x)
    ax.set_xticklabels(labels)
    ax.legend()

    def autolabel(rects):
        for rect in rects:
            height = rect.get_height()
            ax.annotate('{}'.format(height),
                        xy=(rect.get_x() + rect.get_width() / 2, height),
                        xytext=(0, 2),  # 2 points vertical offset
                        textcoords="offset points",
                        ha='center', va='bottom')
    autolabel(rects1)
    autolabel(rects2)
    fig.tight_layout()
    plt.show()

def plot_tag_feedback(tag=""):
    post_filter = {'@CreationDate' : {'$gt':'2019'},
                   '@Tags' : tag
                  }
    posts = list(dbm.posts.find(post_filter))
    print("Posts in total:", len(posts))
    max_weeks = 22
    post_cnt = [0 for _ in range(max_weeks)] 
    answers = [0 for _ in range(max_weeks)] 
    favorite = [0 for _ in range(max_weeks)] 
    for post in posts:
        creation_date = datetime.datetime.strptime(post['@CreationDate'].split('T')[0], '%Y-%m-%d')
        creation_week = creation_date.date().isocalendar()[1]
        
        post_cnt[creation_week -1] += 1
        if '@CommentCount' in post:
            answers[creation_week -1] += int(post['@CommentCount'])
        if '@FavoriteCount' in post:
            favorite[creation_week -1] += int(post['@FavoriteCount'])
       
    labels = ["w{}".format(x) for x in range(1, max_weeks+1)]

    x = np.arange(len(labels))  # the label locations
    width = 0.45  # the width of bars
    
    fig, ax = plt.subplots()
    fig.set_figheight(5)
    fig.set_figwidth(15)
    rects1 = ax.bar(x - width/2, post_cnt, width/2, label='Num. of posts')
    rects2 = ax.bar(x , answers, width/2, label='AnswerCount')
    rects3 = ax.bar(x + width/2, favorite, width/2, label='FavoriteCount')

    # Add some text for labels, title and custom x-axis tick labels, etc.
    ax.set_ylabel('# of posts')
    ax.set_title("Feedback related to the tag '{}'".format(tag))
    ax.set_xticks(x)
    ax.set_xticklabels(labels)
    ax.legend()

    def autolabel(rects):
        for rect in rects:
            height = rect.get_height()
            ax.annotate('{}'.format(height),
                        xy=(rect.get_x() + rect.get_width() / 2, height),
                        xytext=(0, 2),  # 3 points vertical offset
                        textcoords="offset points",
                        ha='center', va='bottom')
    autolabel(rects1)
    autolabel(rects2)
    autolabel(rects3)
    fig.tight_layout()
    plt.show()

Potential Use Cases:¶


Use Case 1: Question Recommender System¶

Description:

  • Recommend questions to experts/users who have a better chance to know a better answer.

StackExchange question-related rules:

  • User should only ask practical, answerable questions.
  • No Open-Ended questions allowed.
  • Every user can ask questions regardless of their reputation

Basic Methodology:

  • Represent every expert user with her set of answers/questions/comments.
  • Run content-similarity techniques between the question's content and each expert's set.


Use Case 2: Trends For You¶

Description:

  • Recommend topics (i.e tags) that are getting increasing attention.

StackExchange tag-related rules:

  • A tag is a word or phrase that describes the topic of the question.
  • Tags are a means of connecting experts with questions.
  • New users are not allowed to create new tags.
  • Each question may only contain 5 tags max.
  • Users may suggest edits or directly edit the tags of a question.

Basic Methodology:

  • Train a machine learning classifier to detect tag-hypes based on a specific tag's popularity trend-line
